scalar DataTime

input GetuserInput {
  after: String
  first: Float
}

type PageInfo {
  endCursor: String
  hasNextPage: Boolean!
  hasPreviousPage: Boolean!
  startCursor: String
}

type Query {
  _service: _Service
  user(data: GetuserInput): userConnection!
}

type _Service {
  sdl: String
}

type user {
  createdAt: DataTime!
  data: [String!]!
  id: String!
  versionId: String!
}

type userConnection {
  edges: [userEdge!]!
  pageInfo: PageInfo!
  totalCount: Float!
}

type userEdge {
  cursor: String!
  node: user!
}
