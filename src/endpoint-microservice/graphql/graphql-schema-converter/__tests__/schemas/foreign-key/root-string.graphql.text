scalar DataTime

scalar JSON

input ProjectGetPostsInput {
  after: String
  first: Float
}

input ProjectGetUsersInput {
  after: String
  first: Float
}

type ProjectPageInfo {
  endCursor: String
  hasNextPage: Boolean!
  hasPreviousPage: Boolean!
  startCursor: String
}

type ProjectPost {
  name: String!
}

type ProjectPostsConnection {
  edges: [ProjectPostsEdge!]!
  pageInfo: ProjectPageInfo!
  totalCount: Float!
}

type ProjectPostsEdge {
  cursor: String!
  node: ProjectPostsNode!
}

type ProjectPostsNode {
  createdAt: DataTime!
  createdId: String!
  data: ProjectPost!
  id: String!
  json: JSON
  updatedAt: DataTime!
  versionId: String!
}

type ProjectUsersConnection {
  edges: [ProjectUsersEdge!]!
  pageInfo: ProjectPageInfo!
  totalCount: Float!
}

type ProjectUsersEdge {
  cursor: String!
  node: ProjectUsersNode!
}

type ProjectUsersNode {
  createdAt: DataTime!
  createdId: String!
  data: ProjectPostsNode!
  id: String!
  json: JSON
  updatedAt: DataTime!
  versionId: String!
}

type Query {
  _service: _Service
  post(id: String!): ProjectPostsNode!
  posts(data: ProjectGetPostsInput): ProjectPostsConnection!
  user(id: String!): ProjectUsersNode!
  users(data: ProjectGetUsersInput): ProjectUsersConnection!
}

type _Service {
  sdl: String
}
